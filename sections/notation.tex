% Notation and Formal Framework
\section{Notation and Formal Framework}
\label{sec:notation}

We define the notation used throughout, then give the formal definitions of TanGCE, the $k$-NN MI estimator, and the SAE-based diagnostics.

\paragraph{Language model and encoder.}
Let $\Sigma^*$ denote the set of all strings over a vocabulary $\Sigma$. A language model $p$ defines a distribution over $\Sigma^*$. We write $\mathrm{enc}_\ell : \Sigma^* \to \mathbb{R}^d$ for the hidden representation at layer $\ell$, where $d$ is the hidden dimension. For a fixed layer $\ell$ (omitted when clear from context), the representation of string $s$ is $\mathbf{x} = \mathrm{enc}_\ell(s) \in \mathbb{R}^d$.

\paragraph{Concepts and data.}
A binary concept is defined by a labeling function $\phi : \Sigma^* \to \mathcal{C}$, where $\mathcal{C} = \{0,1\}$. Each sample $i$ has representation $\mathbf{x}_i \in \mathbb{R}^d$ and concept label $y_i = \phi(s_i) \in \mathcal{C}$, with an optional control label $c_i$ (e.g., profession for gender erasure). We write $\mathcal{D} = \{(\mathbf{x}_i, y_i)\}_{i=1}^N$ for the dataset and $\mathcal{D}_c = \{i : y_i = c\}$ for the class-$c$ subset, with $N_c = |\mathcal{D}_c|$.

The class-conditional representation is the random variable $X_c \sim \mathrm{enc}_\ell(s) \mid \phi(s) = c$, i.e., the distribution of hidden states for strings labeled $c$.

\paragraph{Concept scorer.}
A concept scorer $f_\theta : \mathbb{R}^d \to \mathbb{R}$ is a two-layer MLP trained with binary cross-entropy on $\mathcal{D}$ to predict $y$ from $\mathbf{x}$. Its per-sample gradient with respect to the input is $\nabla_{\mathbf{x}} f_\theta(\mathbf{x}_i) \in \mathbb{R}^d$, which points in the direction of steepest increase in concept probability at $\mathbf{x}_i$.

\paragraph{Conventions.}
Index $i$ ranges over samples.
$\|\cdot\|$ denotes the Euclidean norm unless stated otherwise.
$\psi$ denotes the digamma function $\psi(x) = \frac{d}{dx}\ln\Gamma(x)$.
$\langle \cdot \rangle$ denotes the sample average over all $N$ points.

\subsection{Representation Surgery Baselines}
\label{subsec:baselines}

All erasure methods share a common goal: given representations $\mathbf{x}_i$ that encode concept $\phi$, produce modified representations $\tilde{\mathbf{x}}_i$ from which $\phi$ cannot be recovered, while preserving information unrelated to $\phi$. We review the three linear baselines before introducing the nonlinear GCE family.

\paragraph{Linear projection.}
Given a concept direction $\mathbf{w} \in \mathbb{R}^d$ (typically the difference of class-conditional means, $\mathbf{w} = \mathbb{E}[X_1] - \mathbb{E}[X_0]$), linear projection removes the component along $\mathbf{w}$:
\begin{equation}
\label{eq:linear_proj}
\tilde{\mathbf{x}}_i = (\mathbf{I} - \hat{\mathbf{w}}\hat{\mathbf{w}}^\top)\mathbf{x}_i, \qquad \hat{\mathbf{w}} = \mathbf{w} / \|\mathbf{w}\|.
\end{equation}
This removes a single linear direction but leaves all other concept-correlated structure intact.

\paragraph{INLP (Iterative Nullspace Projection).}
INLP (Ravfogel et al., 2020) iteratively trains linear classifiers $\mathbf{w}_1, \ldots, \mathbf{w}_T$ and projects onto the intersection of their nullspaces:
\begin{equation}
\label{eq:inlp}
\mathbf{P}_{\mathrm{INLP}} = \prod_{t=1}^{T} (\mathbf{I} - \hat{\mathbf{w}}_t \hat{\mathbf{w}}_t^\top), \qquad \tilde{\mathbf{x}}_i = \mathbf{P}_{\mathrm{INLP}} \, \mathbf{x}_i.
\end{equation}
Each $\mathbf{w}_t$ is trained on the residual after projecting out $\mathbf{w}_1, \ldots, \mathbf{w}_{t-1}$. INLP extends linear projection to multiple directions but remains limited to linearly decodable concept information.

\paragraph{LEACE (Least-squares Concept Erasure).}
LEACE (Belrose et al., 2023) finds an oblique projection $\mathbf{P} \in \mathbb{R}^{d \times d}$ of rank $|\mathcal{C}|-1$ and a translation $\mathbf{b}$ that minimally modify representations while ensuring no log-linear classifier can recover the concept:
\begin{equation}
\label{eq:leace}
\tilde{\mathbf{x}}_i = \mathbf{P}\mathbf{x}_i + \mathbf{b}.
\end{equation}
LEACE guarantees $\mathbb{E}[\tilde{X}_c] = \mathbb{E}[X]$ and $\mathrm{Var}[\tilde{X}_c] = \mathrm{Var}[\tilde{X}_{c'}]$ for all $c, c' \in \mathcal{C}$, erasing both mean and covariance differences between class-conditional distributions. This provides the strongest linear erasure guarantee among the baselines.

\subsection{Gradient-Based Concept Erasure}
\label{subsec:gce}

The linear baselines above remove linearly decodable concept information but leave nonlinear structure intact. GCE addresses this by computing per-sample erasure directions from the gradient of a nonlinear concept scorer.

\paragraph{Ambient GCE.}
Given a trained scorer $f_\theta$, the ambient erasure direction at sample $i$ is the unit gradient:
\begin{equation}
\label{eq:ambient_direction}
\mathbf{u}_i = \frac{\nabla_{\mathbf{x}} f_\theta(\mathbf{x}_i)}{\|\nabla_{\mathbf{x}} f_\theta(\mathbf{x}_i)\|}.
\end{equation}
The erased representation is obtained by projecting out the concept component:
\begin{equation}
\label{eq:ambient_erase}
\tilde{\mathbf{x}}_i = \mathbf{x}_i - \lambda \, (\mathbf{x}_i^\top \mathbf{u}_i) \, \mathbf{u}_i,
\end{equation}
where $\lambda > 0$ controls the erasure strength. At $\lambda = 1$, this removes the full projection onto $\mathbf{u}_i$; at $\lambda < 1$, it partially attenuates.

\paragraph{Local tangent space estimation.}
For each query point $\mathbf{x}_i$, we estimate the tangent space $T_{\mathbf{x}_i}\mathcal{M}$ of the data manifold $\mathcal{M}$ via local PCA on its $k$-nearest neighbors.
Let $\mathcal{N}_k(\mathbf{x}_i) = \{\mathbf{x}_{j_1}, \ldots, \mathbf{x}_{j_k}\}$ denote the $k$ nearest neighbors of $\mathbf{x}_i$ in the training set.
We form secant vectors $\mathbf{s}_{j_m} = \mathbf{x}_{j_m} - \mathbf{x}_i$ and compute the local covariance:
\begin{equation}
\label{eq:local_cov}
\mathbf{C}_i = \frac{1}{k} \sum_{m=1}^{k} \mathbf{s}_{j_m} \mathbf{s}_{j_m}^\top \in \mathbb{R}^{d \times d}.
\end{equation}
The top $r$ eigenvectors of $\mathbf{C}_i$ form the tangent basis $\mathbf{B}_i = [\mathbf{b}_i^{(1)}, \ldots, \mathbf{b}_i^{(r)}] \in \mathbb{R}^{d \times r}$, where the intrinsic dimension $r$ is determined automatically by retaining eigenvectors that explain a cumulative fraction of the variance.
The orthogonal projection onto $T_{\mathbf{x}_i}\mathcal{M}$ is:
\begin{equation}
\label{eq:tangent_proj}
\mathbf{P}_{T_i} = \mathbf{B}_i \mathbf{B}_i^\top.
\end{equation}

\paragraph{TanGCE.}
TanGCE projects the ambient erasure direction onto the local tangent space:
\begin{equation}
\label{eq:tangent_direction}
\mathbf{u}_i^{\mathrm{tan}} = \mathbf{P}_{T_i} \, \mathbf{u}_i = \mathbf{B}_i (\mathbf{B}_i^\top \mathbf{u}_i).
\end{equation}
The projected direction $\mathbf{u}_i^{\mathrm{tan}}$ is \emph{not} re-normalized to unit length. Its norm $\|\mathbf{u}_i^{\mathrm{tan}}\| \leq 1$ acts as a per-sample confidence weight: when the concept gradient is mostly off-manifold, $\|\mathbf{u}_i^{\mathrm{tan}}\| \ll 1$, and the effective erasure step is attenuated. The erased representation is:
\begin{equation}
\label{eq:tangent_erase}
\tilde{\mathbf{x}}_i = \mathbf{x}_i - \lambda \, (\mathbf{x}_i^\top \mathbf{u}_i^{\mathrm{tan}}) \, \mathbf{u}_i^{\mathrm{tan}}.
\end{equation}
The effective step size is $\lambda \|\mathbf{u}_i^{\mathrm{tan}}\|^2$, which is maximized when the gradient lies entirely in the tangent space and vanishes when it is entirely off-manifold.

\paragraph{Why tangent projection helps.}
Ambient GCE (Eq.~\ref{eq:ambient_erase}) displaces $\mathbf{x}_i$ along $\mathbf{u}_i$, which may point off the data manifold.
At high $\lambda$, this pushes representations into low-density regions where the displacement magnitude itself correlates with concept signal strength, creating spurious concept information (Section~\ref{sec:lambda_sweep}).
TanGCE constrains the displacement to the tangent space, ensuring that the modified representation stays on (or near) $\mathcal{M}$.

\subsection{Mutual Information Evaluation}
\label{subsec:mi_eval}

We estimate $I(\mathbf{X}; y)$ using the Ross/Gao $k$-nearest-neighbor estimator for mutual information between a continuous variable $\mathbf{X} \in \mathbb{R}^d$ and a discrete variable $y$.

\paragraph{Procedure.}
For each sample $i$:
\begin{enumerate}
    \item Find the $k$-th nearest neighbor of $\mathbf{x}_i$ \emph{within its class} $c(i) = y_i$. Let $\varepsilon_i$ be the Euclidean distance to this neighbor.
    \item Count $m_i$: the number of points from \emph{all classes} within distance $\varepsilon_i$ of $\mathbf{x}_i$ (excluding $\mathbf{x}_i$ itself).
\end{enumerate}
The MI estimate is:
\begin{equation}
\label{eq:knn_mi_formal}
\hat{I}(\mathbf{X}; y) = \psi(k) - \bigl\langle \psi(m_i) \bigr\rangle + \psi(N) - \bigl\langle \psi(N_{c(i)}) \bigr\rangle,
\end{equation}
where $\psi$ is the digamma function, $N_{c(i)}$ is the number of samples in the class of point $i$, and $\langle \cdot \rangle$ averages over all $N$ samples.

\paragraph{Interpretation.}
The estimator measures concept information through local neighborhood structure.
The term $\psi(m_i) - \psi(k)$ captures the local class purity: when same-class points cluster ($m_i \approx k$, mostly same-class neighbors within $\varepsilon_i$), this term is small and MI is high.
When classes are geometrically mixed ($m_i \gg k$, many cross-class neighbors within $\varepsilon_i$), this term is large and MI drops.
The terms $\psi(N) - \psi(N_{c(i)})$ correct for class imbalance.

\paragraph{Erasure evaluation.}
We compute MI on both clean and erased representations:
\begin{itemize}
    \item $\hat{I}(\mathbf{X}; y) - \hat{I}(\tilde{\mathbf{X}}; y) > 0$: concept information was removed.
    \item $\hat{I}(\mathbf{X}; c) \approx \hat{I}(\tilde{\mathbf{X}}; c)$: control-task information was preserved.
\end{itemize}
A method that reduces concept MI while preserving control MI achieves selective erasure.

\subsection{Unsupervised SAE Diagnostics}
\label{subsec:sae_diagnostics}

We train a TopK sparse autoencoder (SAE) on a general-purpose corpus (WikiText), independent of any concept labels.
The SAE maps representations to sparse feature activations: $\mathbf{z} = \mathrm{TopK}(\mathbf{W}_{\mathrm{enc}}(\mathbf{x} - \mathbf{b}_{\mathrm{dec}}) + \mathbf{b}_{\mathrm{enc}})$, where $\mathbf{z} \in \mathbb{R}^{D}$ with $D = \eta d$ ($\eta$ is the expansion factor) and at most $k$ entries are nonzero.

\paragraph{Feature--concept correlation.}
For each SAE feature $j \in \{1, \ldots, D\}$ active on the clean data ($\bar{z}_j > \tau_{\mathrm{act}}$), we compute the point-biserial correlation between its activations and the concept label:
\begin{equation}
\label{eq:point_biserial}
r_j = \frac{\bar{z}_j^{(1)} - \bar{z}_j^{(0)}}{\hat{\sigma}_{z_j}} \cdot \sqrt{\frac{N_0 N_1}{N^2}},
\end{equation}
where $\bar{z}_j^{(c)}$ is the mean activation of feature $j$ for class $c$ and $\hat{\sigma}_{z_j}$ is the pooled standard deviation.
Features with $|r_j| > \tau_{\mathrm{corr}}$ are classified as \emph{concept features}; the remainder are \emph{non-concept features}.

\paragraph{Disruption.}
A feature $j$ is \emph{disrupted} if its relative activation change exceeds a threshold:
\begin{equation}
\label{eq:disruption}
\frac{|\bar{z}_j^{\mathrm{erased}} - \bar{z}_j^{\mathrm{clean}}|}{\bar{z}_j^{\mathrm{clean}}} > \delta,
\end{equation}
where $\bar{z}_j^{\mathrm{clean}}$ and $\bar{z}_j^{\mathrm{erased}}$ are mean activations on clean and erased data, and $\delta$ is the disruption threshold.

\paragraph{Metrics.}
Let $\mathcal{F}$ and $\mathcal{F}^c$ denote the concept and non-concept feature sets.
\begin{align}
\text{Concept removal rate} &= \frac{|\{j \in \mathcal{F} : j \text{ disrupted}\}|}{|\mathcal{F}|}, \label{eq:removal_rate} \\
\text{Damage rate} &= \frac{|\{j \in \mathcal{F}^c : j \text{ disrupted}\}|}{|\mathcal{F}^c|}. \label{eq:damage_rate}
\end{align}

\paragraph{Feature redistribution.}
We recompute point-biserial correlations on the \emph{erased} data.
Features that were non-concept on clean data but become concept-correlated ($|r_j^{\mathrm{erased}}| > \tau_{\mathrm{corr}}$) on erased data indicate concept information \emph{redistribution} rather than removal:
\begin{equation}
\label{eq:redistribution}
\text{Redistribution score} = \frac{|\{j \in \mathcal{F}^c : |r_j^{\mathrm{erased}}| > \tau_{\mathrm{corr}}\}|}{|\mathcal{F}^c|}.
\end{equation}
A method with high removal, low damage, and zero redistribution achieves surgical concept erasure.
